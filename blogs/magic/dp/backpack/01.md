## 基础版

这是最基础的01背包模板
```c++
#include<iostream>
#include<algorithm>
using namespace std;

const int N = 1e3 + 5;

int v[N],w[N];
int f[N][N];
int main()
{
    int n , m ;
    cin >> n >> m;
    for(int i = 1 ; i <= n ; i ++)
        cin >> v[i] >> w[i];
        
    for(int i = 1 ; i <= n ; i ++)
        for(int j = 1 ; j <= m ; j ++)
        {
            f[i][j] = f[i - 1][j];
            if(j >= v[i])
                f[i][j] = max(f[i - 1][j] , f[i - 1][j - v[i]] + w[i]);
        }
    
    cout << f[n][m];
    
    return 0;
}
```
## 一维优化版本

这里是一维优化版本的01背包模板

```c++
#include<iostream>
#include<algorithm>
using namespace std;
const int N = 1005;
int v[N] , w[N] , f[N];

int main()
{
    int m , n;
    cin >> m >> n;
    for(int i = 0 ; i < n ; i ++)
        cin >> v[i] >> w[i];
        
    for(int i = 0 ; i < n ; i ++ )
    {
        for(int j = m ; j >= v[i] ; j --)           //为什么从后往前遍历体积，具体看01背包的一维版本。如果从前往后，
            f[j] = max(f[j] , f[j - v[i]] + w[i]);  //则会出现上一个状态还未产生 ， 反正是从二维版本优化而来的。
    }
    
    cout << f[m];
    
    return 0;
}
```

## 不求价值，求方案数量

```c++
#include<iostream>
#include<algorithm>
using namespace std;
const int N = 10005;
int f[N];

int main()
{
    int n , m ;
    cin >> n >> m;
    f[0] = 1;                               //f[m]的状态是由f[0]转移过来的。
    for(int i = 0 ; i < n ; i ++)
    {
        int v;
        cin >> v;
        for(int j = m ; j >= v ; j --)
            f[j] += f[j - v];               //当枚举到j = v 时，f[v] += f[0],体积为v的时候就选一个这个刚好填满v的体积
    }                                       //所以是一种情况,在这个基础上，f[v + x] += f[x],就会在原来取一个v的基础上再取一个x
    cout << f[m];                           //推广一下，f[v]和f[x]有可能是由其他情况转移而来的,那么循环结束以后，f[m]存的就是
                                            //所有能够组成体积为m时的种数
    return 0;
}
```
